Task 03

Panoda sagda liste olusturun

pos=bottom parametresi ile sagda liste olusturma ve name degerini Right olarak gonderme.

Request

Method
POST
URL
https://api.trello.com/1/lists?name={right}&idBoard={{idboard}}&key={{apikey}}&token={{token}}&pos=bottom

Test Script

pm.test("Status code is 200", function () {
    pm.response.to.have.status(200);
});

pm.test("Istek suresi 500 ms'den kisa olmali", function () {
    pm.expect(pm.response.responseTime).to.be.below(500);
});

pm.test("name bir string olmalidir", function () {
    var jsonData = pm.response.json();
    pm.expect(jsonData.name).to.be.a("string");
});
